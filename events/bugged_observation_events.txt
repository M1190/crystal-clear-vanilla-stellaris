namespace = observation
# Societal Shift towards Competitive
country_event = {
	id = observation.0001
	hide_window = yes
	trigger = {
		is_country_type = primitive
		NOT = {
			has_ethic = ethic_fanatic_competitive
		}
		capital_scope = {
			has_observation_outpost = yes
			observation_outpost = {
				has_mission = native_indoctrination
			}
			observation_outpost_owner = {
				OR = {
					has_ethic = ethic_fanatic_competitive
					has_ethic = ethic_competitive
				}
			}
			any_owned_pop = {
				has_ethic = ethic_competitive
			}
			OR = {
				root = {
					NOT = {
						has_ethic = ethic_competitive
					}
				}
				count_owned_pop = {
					limit = {
						has_ethic = ethic_competitive
					}
					count > 1
				}
			}
		}
	}
	mean_time_to_happen = {
		months = 12
	}
	immediate = {
		shift_ethic = "ethic_competitive"
		capital_scope = {
			observation_outpost_owner = {
				country_event = {
					id = observation.0002
				}
			}
		}
	}
}

country_event = {
	id = observation.0002
	title = "observation.0002.name"
	desc = "observation.0002.desc"
	picture = GFX_evt_medieval_alien_civilization
	show_sound = event_primitive_civilization
	location = from
	is_triggered_only = yes
	option = {
		name = "observation.0002.a"
	}
}
