# Global flag; oher mods can use this

event = {
	id = emex_update.1
	hide_window = yes
	fire_only_once = yes
	is_triggered_only = yes
	
	trigger = { NOT = { has_global_flag = emex_active }}
	
	immediate = {
		if = {
			limit = {
				NOT = { has_global_flag = emex_active }
			}
			set_global_flag = emex_active
		}
	}
}
