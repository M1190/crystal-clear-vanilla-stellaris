country_event = {
	id = opcivicsb.1
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		if = { 																	
			limit = { 
				has_civic = civic_legendary_fleet 
			} 
			every_owned_leader = {												
                limit = {
                    leader_class = admiral
               }
                add_trait = leader_trait_supreme_tactician
				}
			}
		}
}
