raiding = {
	trigger = {
		OR = {
			owner = {
				has_ascension_perk = ap_nihilistic_acquisition
			}
			owner = {
				has_valid_civic = civic_barbaric_despoilers
			}
			owner = {
				has_valid_civic = civic_living_hosts
			}
			AND = {
				owner = {
					is_country_type = enclave_mercenary
				}
				controller = {
					OR = {
						has_ascension_perk = ap_nihilistic_acquisition
						has_valid_civic = civic_barbaric_despoilers
						has_valid_civic = civic_living_hosts
					}
				}
			}
		}
	}
	default = no
	stop_when_armies_dead = no
	abduct_pops = yes
	planet_damage = 1
	army_damage = 0.5
	kill_pop_chance = {
		base = 1
	}
	min_pops_to_kill_pop = 2
	# root = fleet
	# from = planet
	ai_weight = {
		weight = 8
		modifier = {
			factor = 0
			exists = from
			from = {
				owner = {
					NOT = {
						is_hostile = root.owner
					}
				}
			}
		}
		modifier = {
			factor = 0.01
			exists = from
			from = {
				owner = {
					has_claim = root.solar_system
				}
			}
		}
	}
}

liberation = {
	trigger = {
		OR = {
			owner = {
				has_ascension_perk = ap_eternal_freedom
			}
			owner = {
				has_valid_civic = civic_machine_servitor
			}
			owner = {
				has_valid_civic = civic_emancipators
			}
			owner = {
				has_valid_civic = civic_broken_shackles
			}
			AND = {
				owner = {
					is_country_type = enclave_mercenary
				}
				controller = {
					OR = {
						has_ascension_perk = ap_eternal_freedom
						has_valid_civic = civic_machine_servitor
						has_valid_civic = civic_emancipators
						has_valid_civic = civic_broken_shackles
					}
				}
			}
		}
	}
	default = no
	stop_when_armies_dead = no
	abduct_pops = yes
	planet_damage = 1
	army_damage = 0.5
	kill_pop_chance = {
		# Should only target oppressed pops
		base = 1.25
		modifier = {
			factor = 0
			OR = {
				is_sapient = no
				is_robot_pop = yes
				pop_has_trait = trait_hive_mind
				NOR = {
					is_enslaved = yes
					is_being_purged = yes
					has_job = necro_apprentice
					AND = {
						OR = {
							is_being_assimilated = yes
							has_trait = trait_cybernetic
						}
						exists = owner
						owner = {
							OR = {
								has_valid_civic = civic_infestation
								has_valid_civic = civic_machine_assimilator
							}
						}
					}
				}
			}
		}
	}
	min_pops_to_kill_pop = 2
	# root = fleet
	# from = planet
	ai_weight = {
		weight = 8
		modifier = {
			factor = 0
			exists = from
			from = {
				owner = {
					NOT = {
						is_hostile = root.owner
					}
				}
			}
		}
		modifier = {
			factor = 0.01
			exists = from
			from = {
				owner = {
					has_claim = root.solar_system
				}
			}
		}
	}
}

death = {
	trigger = {
		OR = {
			owner = {
				has_technology = tech_reanimated_bombardment
				OR = {
					NOT = {
						is_country_type = default
					}
					has_policy_flag = orbital_bombardment_indiscriminate
					has_policy_flag = orbital_bombardment_armageddon
				}
			}
			AND = {
				owner = {
					is_country_type = enclave_mercenary
				}
				controller = {
					has_technology = tech_reanimated_bombardment
					OR = {
						NOT = {
							is_country_type = default
						}
						has_policy_flag = orbital_bombardment_indiscriminate
						has_policy_flag = orbital_bombardment_armageddon
					}
				}
			}
		}
	}
	default = no
	stop_when_armies_dead = no
	abduct_pops = no
	planet_damage = 0.5
	army_damage = 1.5
	kill_pop_chance = {
		# Should only target organics
		base = 1
		modifier = {
			factor = 0
			OR = {
				pop_has_trait = trait_machine_unit
				pop_has_trait = trait_mechanical
			}
		}
	}
	min_pops_to_kill_pop = 0
	# root = fleet
	# from = planet
	ai_weight = {
		weight = 10
		modifier = {
			factor = 0.01
			exists = from
			from = {
				owner = {
					NOT = {
						is_hostile = root.owner
					}
				}
			}
		}
	}
}
