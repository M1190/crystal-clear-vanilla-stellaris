### Federation Builders ###
empire_name_format = {
	random_weight = {
		factor = 0
		modifier = {
			add = 5
			OR = {
				has_authority = auth_democratic
				has_authority = auth_oligarchic
			}
			NOR = {
				has_ethic = "ethic_xenophobe"
				has_ethic = "ethic_fanatic_xenophobe"
				has_ethic = "ethic_authoritarian"
				has_ethic = "ethic_fanatic_authoritarian"
				has_ethic = "ethic_militarist"
				has_ethic = "ethic_fanatic_militarist"
			}
			OR = {
				has_ethic = "ethic_pacifist"
				has_ethic = "ethic_fanatic_pacifist"
				has_ethic = "ethic_xenophile"
				has_ethic = "ethic_fanatic_xenophile"
				has_origin = origin_league
			}
			OR = {
				has_ethic = ethic_egalitarian
				has_ethic = ethic_fanatic_egalitarian
				has_valid_civic = civic_beacon_of_liberty
				has_valid_civic = civic_idealistic_foundation
				has_valid_civic = civic_free_haven
				has_valid_civic = civic_parliamentary_system
				has_origin = origin_league
			}
			is_pirate = no
			is_primitive = no
			NOT = { is_country_type = fallen_empire }
			NOT = { is_country_type = awakened_fallen_empire }
		}
	}
	lookups = "<dem_federation_builder_pre_desc> <dem_federation_builder> of [This.Capital.GetName]"
	format = format.dem_federation_builder.1
	prefix_format = format_prefix.dem_federation_builder.1
	noun = format.homeworld
	# United Nations of Earth
}

### Agrarian Idyll ###
# Imperial / Dictatorial Idyll
empire_name_format = {
	random_weight = {
		factor = 0
		modifier = {
			add = 2
			has_valid_civic = civic_agrarian_idyll
			OR = {
				has_authority = auth_imperial
				has_authority = auth_dictatorial
			}
			is_plantoid_empire = no
			is_pirate = no
			is_primitive = no
			NOT = { is_country_type = fallen_empire }
			NOT = { is_country_type = awakened_fallen_empire }
		}
		modifier = {
			add = 5
			has_valid_civic = civic_agrarian_idyll
			OR = {
				has_authority = auth_imperial
				has_authority = auth_dictatorial
			}
			is_plantoid_empire = yes
			is_pirate = no
			is_primitive = no
			NOT = { is_country_type = fallen_empire }
			NOT = { is_country_type = awakened_fallen_empire }
		}
	}
	lookups = "[This.GetSpeciesName] <imperial_idyll>"
	format = format.imp_idyll.1
	prefix_format = format.imp_idyll.1
	noun = format.species
}
