decision_cult_world = {
	owned_planets_only = yes

	enactment_time = 60
	resources = {
		category = decisions
		cost = {
			unity = 1000
		}
	}

	potential = {
		is_capital = no
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = slave_colony }
		NOT = { has_modifier = industrial_core }
		NOT = { has_modifier = galactic_refuse_centre }
		NOT = { has_modifier = research_complex }
		owner = {
			OR = {
				has_ascension_perk = ap_consecrated_worlds
				has_ascension_perk = ap_transcendence
				has_valid_civic = civic_astral_obsession
			}
		}
		is_artificial = no
		NOT = {
			owner = {
				any_owned_planet = { has_modifier = cult_world }
			}
		}
	}

	allow = {
		planet_size >= 9
	}

	effect = {
		add_modifier = {
			modifier = cult_world
			days = -1
		}
		hidden_effect = { validate_planet_buildings_and_districts = yes }
	}

	ai_weight = {
		weight = 0
	}
}
