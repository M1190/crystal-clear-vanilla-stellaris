decision_cpb_sell_planet = {
	owned_planets_only = yes
	enactment_time = 90
	sound = event_galactic_market
	resources = {
		category = decisions
		cost = {
			influence = 25
		}
	}
	potential = {
		owner = { is_regular_empire = yes }
		NOT = { has_modifier = cpb_sold_planet }
		owner = { has_civic = cpb_civic_corporate_directorate }
	}
	effect = {
		hidden_effect = {
			add_modifier = { modifier = "cpb_sold_planet" days = -1 }
		}
		owner = {
			add_monthly_resource_mult = {
				resource = energy
				value = 5
				min = @tier5materialmin
				max = @tier5materialmax
			}
			add_monthly_resource_mult = {
				resource = unity
				value = 3
				min = @tier3materialmin
				max = @tier3materialmax
			}
		}
		custom_tooltip = decision_cpb_sell_planet_effects_custom
	}
	allow = {
		NOT = { is_capital = yes }
	}	
	ai_weight = {
		weight = 100
	}
}
decision_cpb_create_bank_world = {
	owned_planets_only = yes
	enactment_time = 180
	resources = {
		category = decisions
		cost = {
			influence = 100
		}
	}
	potential = {
		NOT = { has_modifier = resort_colony }
		NOT = { has_modifier = penal_colony }
		NOT = { has_modifier = slave_colony }
		NOT = { has_modifier = cpb_bank_world_colony }
		owner = {
			is_regular_empire = yes
			has_civic = cpb_civic_megabank
		}
		is_artificial = no
	}
	effect = {
		hidden_effect = {
			add_modifier = { modifier = "cpb_bank_world_colony" days = -1 }
			remove_building = building_colony_shelter
			add_building = cpb_building_security_vault
			validate_planet_buildings_and_districts = yes
			add_deposit = cpb_vault_tunnel_systems
		}
		custom_tooltip = decision_cpb_create_bank_world_effects_custom
	}
	allow = {
		NOT = { is_capital = yes }
		custom_tooltip = {
			fail_text = decision_resort_colony_no_districts
			num_districts = { type = any value < 1 }
			num_buildings = { type = any value < 2 }
		}
		custom_tooltip = {
			fail_text = no_upgraded_capital_tooltip
			has_upgraded_capital = no
		}
		custom_tooltip = {
			fail_text = cpb_market_has_been_founded
			has_global_flag = galactic_market_founded
		}
	}	
	ai_weight = {
		weight = 1
	}
}
