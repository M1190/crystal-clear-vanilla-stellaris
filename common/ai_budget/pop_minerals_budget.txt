minerals_expenditure_decisions = {
	resource = minerals
	type = expenditure
	category = decisions

	potential = {
		# only consider these kinds of decisions when "wealthy"
		has_resource = {
			type = minerals
			amount > 5000
		}
	}

	fraction = {
		weight = 0.20
	}

	# based on mastery of nature decision
	static_max = {
		base = 20000
	}
}

minerals_expenditure_deposit_blockers = {
	resource = minerals 
	type = expenditure
	category = deposit_blockers
	
	potential = {
		any_owned_planet = {
			num_uncleared_blockers >= 1 # for compatability with mod-added blockers
			num_free_districts = { type = any value < 1 }
		}
	}
	
	fraction = {
		weight = 0.2
		modifier = {
			factor = 6

			# if we have any planet that is currently completely blocked
			any_owned_planet = {
				num_free_districts = { type = any value < 1 }
			}
		}
	}

	# base remove blocker cost
	static_min = {
		base = 250
	}

	static_max = {
		base = 1000
	}
}

minerals_expenditure_colonies_master_computer_expand = {
	resource = minerals
	type = expenditure
	category = colonies

	potential = {
		is_robot_empire = yes
		has_origin = origin_master_computer
		ai_colonize_plans > 0
		has_monthly_income = {
			resource = minerals
			value > 0
		}
	}

	fraction = {
		weight = 0.8
	}

	static_min = {
		base = 500
	}

	static_max = {
		base = 1000
	}
}
