		modifier = {
			factor = 1.25

				area = society
				has_trait = "leader_trait_expertise_new_worlds"
			}
		}
		modifier = {
			factor = 2
			is_xenophobe = yes
		}
		modifier = {
			factor = 5
			OR = {
				any_owned_planet = {
					habitability = {
						who = root
						value < 0.5
					}
				}
				any_system_within_border = {
					any_system_planet = {
						is_planet_class = pc_nuked
					}
				}
			}
		}
		modifier = {
			factor = 5
			has_origin = origin_life_seeded
		}
		modifier = {
			factor = 2
			is_ai = no			#auto-research
		}
	}
}
