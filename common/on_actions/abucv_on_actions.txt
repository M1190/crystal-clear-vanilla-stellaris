# Triggers when the game starts
on_game_start = {
	events = {
		abucv_events.77 # Jeager Titanic Life Planet Modifier
		abucv_events.90 # Doomsayer Variable Set
		abucv_events.11 # Divide by Zero Start
	}
}

# No scope, like on_game_start
on_monthly_pulse = {
	events = {
		abucv_events.30 # AD bombing check
	}
}

# No scope, like on_game_start
on_yearly_pulse = {
	events = {
		abucv_events.95 # Doomsayer YearCheck
	}
}

# Executed as new ruler has been set
# This = Country
on_ruler_set = {
	events = {
		abucv_events.3	# Deadly Politics Kill Ruler Event Start
	}
}

# A pop has finished purging
# This = Planet scope
# From = Country
# FromFrom = Pop
on_pop_purged = {
	events = {
		abucv_events.91 # Gain resources for Doomsayer civic
	}
}
