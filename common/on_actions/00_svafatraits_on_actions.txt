# Triggers when the game starts
on_game_start = {
	events = {
		svafa_traits.1		#svafa traits add modifier event
	}
}

on_modification_complete = {
	events = {
		svafa_traits.1		#svafa traits add modifier event
		svafa_traits.2		#svafa traits remove modifier event
	}
}

on_yearly_pulse = {
	events = {
		svafa_traits.1		#svafa traits add modifier event
		svafa_traits.2		#svafa traits remove modifier event
	}
}
