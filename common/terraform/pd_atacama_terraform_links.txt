	from = "pc_hadesert"
	to = "pc_continental"

	resources = {
		category = terraforming

		cost = {
			energy = @terraforming_cost_level_2
		}
	}

	duration = @terraforming_duration_level_2

	potential = {
		has_technology = "tech_terrestrial_biome_manipulation"
		NOR = {
			has_policy_flag = terraforming_basic_invisible
			has_policy_flag = terraforming_wet_invisible
		}
	}

	condition = {
		has_technology = "tech_terrestrial_biome_manipulation"
	}

	ai_weight = {
		weight = 0
		modifier = {
			weight = 1
			should_terraform_trigger = {
				CLASS_FROM = pc_hadesert
				CLASS_TO = pc_continental
			}
		}
	}
}

	from = "pc_hadesert"
	to = "pc_geothermal"

	resources = {
		category = terraforming

		cost = {
			energy = @terraforming_cost_level_3
			#exotic_gases = @terraforming_cost_level_1
		}
	}

	duration = @terraforming_duration_level_3

	potential = {
		NOR = {
			has_policy_flag = terraforming_rare_invisible
			has_policy_flag = terraforming_wet_invisible
		}
		OR = {
			has_ascension_perk = ap_sublunaris_progressus
			from = {
				has_modifier = pdterra_geothermal
			}
		}
	}
	
	condition = {
		has_technology = tech_climate_restoration
	}
	
	effect = {
		from = {
			remove_modifier = pdterra_geothermal
			#add_modifier = { 
				#modifier = geothermal
				#days = -1
			#}
			add_deposit = d_geothermal_pools_deposit
		}
	}

	ai_weight = {
		weight = 0
		modifier = {
			weight = 1
			should_terraform_trigger = {
				CLASS_FROM = pc_hadesert
				CLASS_TO = pc_geothermal
			}
		}
	}
}
